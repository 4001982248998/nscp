cmake_minimum_required(VERSION 2.6)

SET(TARGET CheckNSCP)

PROJECT(${TARGET})

SET(SRCS
	"${TARGET}.cpp"
	${NSCP_DEF_PLUGIN_CPP}
)

ADD_DEFINITIONS(${NSCP_GLOBAL_DEFINES})

IF(WIN32)
	SET(SRCS ${SRCS}
		"${TARGET}.h"
		"${TARGET}.def"

		${NSCP_DEF_PLUGIN_HPP}
	)
ENDIF(WIN32)


ADD_LIBRARY(${TARGET} MODULE ${SRCS})


target_link_libraries(${TARGET}
	${NSCP_DEF_PLUGIN_LIB}
)
INCLUDE(${BUILD_CMAKE_FOLDER}/module.cmake)
