cmake_minimum_required(VERSION 2.6)

SET(TARGET CheckWMI)
	
PROJECT(${TARGET})

CREATE_MODULE(SRCS ${CMAKE_CURRENT_SOURCE_DIR} ${CMAKE_CURRENT_BINARY_DIR})

SET(SRCS ${SRCS}
	stdafx.cpp
	"${TARGET}.cpp"
	WMIQuery.cpp
	
	${NSCP_DEF_PLUGIN_CPP}
	${NSCP_FILTER_CPP}
)


ADD_DEFINITIONS(${NSCP_GLOBAL_DEFINES})

IF(WIN32)
SET(SRCS ${SRCS}
	stdafx.h
	"${TARGET}.h"
	WMIQuery.h

	${NSCP_DEF_PLUGIN_HPP}
	${NSCP_FILTER_HPP}
)
ENDIF(WIN32)

add_library(${TARGET} MODULE ${SRCS})

target_link_libraries(${TARGET}
	${NSCP_DEF_PLUGIN_LIB}
	${Boost_REGEX_LIBRARY}
	${Boost_PROGRAM_OPTIONS_LIBRARY}
	${NSCP_FILTER_LIB}
	expression_parser
	Wbemuuid.lib
)

INCLUDE(${BUILD_CMAKE_FOLDER}/module.cmake)
