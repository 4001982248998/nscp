cmake_minimum_required(VERSION 2.6)

SET(TARGET LUAScript)
	
PROJECT(${TARGET})

SET(SRCS
	stdafx.cpp
	"${TARGET}.cpp"
	lua_wrappers.cpp

	${NSCP_DEF_PLUGIN_CPP}
)

ADD_DEFINITIONS(${NSCP_GLOBAL_DEFINES})
INCLUDE_DIRECTORIES(${LUA_INCLUE_DIR})

IF(WIN32)
	SET(SRCS ${SRCS}
		stdafx.h
		"${TARGET}.h"
		"${TARGET}.def"
		lua_wrappers.hpp
		script_wrapper.hpp

		${NSCP_DEF_PLUGIN_HPP}
	)
ENDIF(WIN32)

add_library(${TARGET} MODULE ${SRCS})

SET(LUA_PB)
IF (PROTOC_GEN_LUA_FOUND)
	MESSAGE(STATUS "Lua has PB support")
	SET(LUA_PB lua_pb_static)
	ADD_DEFINITIONS(-DHAVE_LUA_PB)
	INCLUDE_DIRECTORIES(${NSCP_PROJECT_BINARY_DIR}/libs/lua_pb)
	INCLUDE_DIRECTORIES(${NSCP_PROJECT_BINARY_DIR}/libs/protobuf)
ENDIF()


target_link_libraries(${TARGET}
	${Boost_FILESYSTEM_LIBRARY}
	${Boost_PROGRAM_OPTIONS_LIBRARY}
	${NSCP_DEF_PLUGIN_LIB}
	${LUA_LIB}
	${LUA_PB}
)
INCLUDE(${BUILD_CMAKE_FOLDER}/module.cmake)
